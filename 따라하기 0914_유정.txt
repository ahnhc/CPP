#include <iostream>
#define PRINT_NAME std::cout << "[이름: 천유정]" << std::endl
#define PRINT_ID std::cout << "[학번: 202322523]" << std::endl
#define PRINT_TIMESTAMP std::cout << "[" << __TIMESTAMP__ << "]\n" << std::endl

using std::cout;
using std::endl;

int main()
{
	PRINT_NAME;
	PRINT_ID;
	PRINT_TIMESTAMP;

	char str1[20] = "Hello, world!";
	char str2[] = "Hello, world!";
	const char* pstr3 = "Hello, world!"; //const

	cout << "size of str1: " << sizeof(str1) << ", " << str1 << endl;
	cout << "size of str2: " << sizeof(str2) << ", " << str2 << endl;
	cout << "size of str3: " << sizeof(pstr3) << ", " << pstr3 << endl;

	return 0;
}
